import express, { type Request, Response, NextFunction } from "express";
import { registerRoutes } from "./routes";
import { setupVite, serveStatic, log } from "./vite";

const app = express();

declare module 'http' {
  interface IncomingMessage {
    rawBody: unknown
  }
}

app.use(express.json({
  limit: '10mb',
  verify: (req, _res, buf) => {
    req.rawBody = buf;
  }
}));
app.use(express.urlencoded({ extended: false, limit: '10mb' }));

// --- CRITICAL: CORS Middleware MUST come before routes ---
app.use((req, res, next) => {
  const origin = req.headers.origin;
  
  // List of allowed origins - MUST include https://localhost for Capacitor
  const allowedOrigins = [
    'https://localhost',        // Capacitor Android
    'http://localhost',
    'capacitor://localhost',    // Capacitor iOS
    'ionic://localhost',
    'https://lyricsensei.com',
    'http://lyricsensei.com',
    'http://localhost:5000',
    'http://localhost:5173',    // Vite dev
    'http://10.0.2.2:5000',     // Android emulator
  ];

  // Check if origin is allowed
  if (origin) {
    const isAllowed = allowedOrigins.includes(origin) || 
                      origin.includes('localhost') || 
                      origin.includes('10.0.2.2') ||
                      origin.startsWith('capacitor://') ||
                      origin.startsWith('ionic://');

    if (isAllowed) {
      log(`✓ CORS allowing origin: ${origin}`);
      res.setHeader('Access-Control-Allow-Origin', origin);
      res.setHeader('Access-Control-Allow-Credentials', 'true');
    } else {
      log(`✗ CORS blocking origin: ${origin}`);
    }
  } else {
    // No origin header - allow with wildcard (for direct requests)
    res.setHeader('Access-Control-Allow-Origin', '*');
  }

  // ALWAYS set these headers for ALL requests
  res.setHeader('Access-Control-Allow-Methods', 'GET, POST, PUT, PATCH, DELETE, OPTIONS, HEAD');
  res.setHeader('Access-Control-Allow-Headers', 'Content-Type, Authorization, Accept, Origin, X-Requested-With');
  res.setHeader('Access-Control-Max-Age', '86400'); // 24 hours

  // Handle preflight requests
  if (req.method === 'OPTIONS') {
    log(`Handling OPTIONS preflight for ${req.path}`);
    return res.status(204).end();
  }

  next();
});

// Request logging
app.use((req, res, next) => {
  const start = Date.now();
  const path = req.path;
  let capturedJsonResponse: Record<string, any> | undefined = undefined;

  const originalResJson = res.json;
  res.json = function (bodyJson, ...args) {
    capturedJsonResponse = bodyJson;
    return originalResJson.apply(res, [bodyJson, ...args]);
  };

  res.on("finish", () => {
    const duration = Date.now() - start;
    if (path.startsWith("/api")) {
      let logLine = `${req.method} ${path} ${res.statusCode} in ${duration}ms`;
      if (capturedJsonResponse) {
        logLine += ` :: ${JSON.stringify(capturedJsonResponse)}`;
      }

      if (logLine.length > 80) {
        logLine = logLine.slice(0, 79) + "…";
      }
      
      log(logLine);
    }
  });

  next();
});

// Register your API routes
(async () => {
  const server = registerRoutes(app);

  // Error handler
  app.use((err: any, _req: Request, res: Response, _next: NextFunction) => {
    const status = err.status || err.statusCode || 500;
    const message = err.message || "Internal Server Error";
    log(`[ERROR] ${status}: ${message}`);
    res.status(status).json({ message });
  });

  // Setup Vite or static serving
  if (app.get("env") === "development") {
    await setupVite(app, server);
  } else {
    serveStatic(app);
  }

  const PORT = process.env.PORT || 5000;
  server.listen(PORT, "0.0.0.0", () => {
    log(`✓ Server running on http://0.0.0.0:${PORT}`);
    log(`✓ CORS enabled for Capacitor mobile apps`);
  });
})();
